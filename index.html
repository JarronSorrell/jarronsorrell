const qPCRCancerPanel = () => {
  // Simulate qPCR amplification data for different cancer biomarkers
  // Each array represents the fluorescence values across cycles for a specific gene target
  const generateAmplificationCurve = (ct, efficiency, baseline) => {
    const data = [];
    for (let cycle = 1; cycle <= 40; cycle++) {
      // Sigmoid curve model for qPCR amplification
      let fluorescence;
      if (cycle < ct - 5) {
        // Baseline region
        fluorescence = baseline + Math.random() * 0.05;
      } else {
        // Exponential and plateau regions
        const exponent = efficiency * (cycle - ct);
        fluorescence = baseline + (100 - baseline) / (1 + Math.exp(-exponent));
        // Add small noise
        fluorescence += (Math.random() - 0.5) * 2;
      }
      data.push({
        cycle,
        fluorescence: Math.max(fluorescence, baseline)
      });
    }
    return data;
  };

  // Generate data for several gene targets enriched with SureSelect
  const geneTargets = [
    { name: 'BRCA1', ct: 22.3, efficiency: 1.2, baseline: 3.2, color: '#8884d8' },
    { name: 'BRCA2', ct: 23.1, efficiency: 1.15, baseline: 3.4, color: '#82ca9d' },
    { name: 'TP53', ct: 18.7, efficiency: 1.3, baseline: 3.0, color: '#ff7300' },
    { name: 'KRAS', ct: 20.9, efficiency: 1.25, baseline: 3.3, color: '#0088fe' },
    { name: 'EGFR', ct: 24.6, efficiency: 1.1, baseline: 3.5, color: '#ff3399' },
    { name: 'PIK3CA', ct: 25.2, efficiency: 1.05, baseline: 3.6, color: '#8dd1e1' },
    { name: 'NTC', ct: 38.5, efficiency: 0.8, baseline: 3.1, color: '#a4a4a4' }, // No template control
  ];

  // Create amplification curves for each target
  const amplificationData = {};
  geneTargets.forEach(target => {
    amplificationData[target.name] = generateAmplificationCurve(target.ct, target.efficiency, target.baseline);

Project Overview

This visualization demonstrates the analysis of a cancer panel test using SureSelect target enrichment technology. The graph displays qPCR amplification curves for multiple cancer-related genes (BRCA1, BRCA2, TP53, KRAS, EGFR, and PIK3CA) along with a no-template control.

Technologies Used

  • React - For building the interactive user interface
  • Recharts - For creating responsive and customizable charts
  • Tailwind CSS - For styling components without writing custom CSS

Technical Highlights

  • Simulation of sigmoid qPCR amplification curves with appropriate baseline, exponential, and plateau phases
  • Implementation of threshold cycle (Ct) calculation using linear interpolation
  • Dynamic data visualization with interactive tooltips and legend
  • Responsive design that works across different screen sizes
